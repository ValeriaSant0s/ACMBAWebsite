<script>
(function () {
  const canvas = document.getElementById("gameCanvas");
  const ctx = canvas.getContext("2d");
 
  const startMenu = document.getElementById("startMenu");
  const blackScreen = document.getElementById("blackScreen");
  const gameArea = document.getElementById("gameArea");
  const joinBtn = document.getElementById("joinBtn");
  const exitBtn = document.getElementById("exitBtn");
  const gameExitBtn = document.getElementById("gameExitBtn");
  const playerNameInput = document.getElementById("playerNameInput");
  const inventoryDiv = document.getElementById("inventory");
  const selectedBlockLabel = document.getElementById("selectedBlockLabel");
 
  const tileSize = 32;
  const cols = canvas.width / tileSize;
  const rows = canvas.height / tileSize;
 
  let gameStarted = false;
  let selectedBlock = "grass";
 
  const blocks = {
	air:   { name: "Air",   color: "#bfe9ff", solid: false },
	grass: { name: "Grass", color: "#56b84a", solid: true },
	dirt:  { name: "Dirt",  color: "#9b6a3c", solid: true },
	stone: { name: "Stone", color: "#8a8f98", solid: true },
	wood:  { name: "Wood",  color: "#8b5a2b", solid: true },
	leaf:  { name: "Leaf",  color: "#3f9f46", solid: true },
	water: { name: "Water", color: "#3ba7ff", solid: false }
  };
 
  const inventory = {
	grass: 5,
	dirt: 5,
	stone: 3,
	wood: 3,
	leaf: 2
  };
 
  const world = [];
 
  const player = {
	x: 2,
	y: 8,
	color: "#2937ff",
	name: "Player"
  };
 
  function createWorld() {
	for (let y = 0; y < rows; y++) {
  	world[y] = [];
 
  	for (let x = 0; x < cols; x++) {
    	if (y < 7) {
      	world[y][x] = "air";
    	} else if (y === 7) {
      	world[y][x] = "grass";
    	} else if (y < 10) {
      	world[y][x] = "dirt";
    	} else {
      	world[y][x] = Math.random() > 0.25 ? "stone" : "dirt";
    	}
  	}
	}
 
	world[6][11] = "wood";
	world[5][11] = "wood";
	world[4][11] = "wood";
	world[3][10] = "leaf";
	world[3][11] = "leaf";
	world[3][12] = "leaf";
	world[4][10] = "leaf";
	world[4][12] = "leaf";
 
	world[7][15] = "water";
	world[7][16] = "water";
	world[7][17] = "water";
 
    world[player.y][player.x] = "air";
  }
 
  function drawBlock(x, y, type) {
	const block = blocks[type];
 
	ctx.fillStyle = block.color;
	ctx.fillRect(x * tileSize, y * tileSize, tileSize, tileSize);
 
	if (type !== "air") {
  	ctx.strokeStyle = "rgba(0, 0, 0, 0.12)";
  	ctx.strokeRect(x * tileSize, y * tileSize, tileSize, tileSize);
	}
 
	if (type === "grass") {
  	ctx.fillStyle = "#77d967";
  	ctx.fillRect(x * tileSize, y * tileSize, tileSize, 7);
	}
 
	if (type === "stone") {
  	ctx.fillStyle = "rgba(255,255,255,0.12)";
  	ctx.fillRect(x * tileSize + 6, y * tileSize + 7, 10, 4);
  	ctx.fillRect(x * tileSize + 18, y * tileSize + 20, 8, 4);
	}
 
	if (type === "wood") {
  	ctx.fillStyle = "rgba(255,255,255,0.18)";
  	ctx.fillRect(x * tileSize + 8, y * tileSize + 4, 4, 24);
  	ctx.fillRect(x * tileSize + 20, y * tileSize + 4, 3, 24);
	}
 
	if (type === "water") {
  	ctx.fillStyle = "rgba(255,255,255,0.22)";
  	ctx.fillRect(x * tileSize + 4, y * tileSize + 9, 24, 4);
	}
  }
 
  function drawPlayerName(px, py) {
	const name = player.name || "Player";
 
	ctx.font = "bold 12px Arial";
	ctx.textAlign = "center";
	ctx.textBaseline = "bottom";
 
	const textWidth = ctx.measureText(name).width;
	const centerX = px + tileSize / 2;
	const nameY = py - 4;
 
	ctx.fillStyle = "rgba(255, 255, 255, 0.82)";
    ctx.fillRect(centerX - textWidth / 2 - 6, nameY - 16, textWidth + 12, 16);
 
	ctx.fillStyle = "#1f2a44";
	ctx.fillText(name, centerX, nameY - 2);
  }
 
  function drawPlayer() {
	const px = player.x * tileSize;
	const py = player.y * tileSize;
 
	drawPlayerName(px, py);
 
	ctx.fillStyle = player.color;
	ctx.fillRect(px + 7, py + 7, 18, 20);
 
	ctx.fillStyle = "#9b87ff";
	ctx.fillRect(px + 9, py + 3, 14, 10);
 
	ctx.fillStyle = "#ffffff";
	ctx.fillRect(px + 12, py + 7, 3, 3);
	ctx.fillRect(px + 18, py + 7, 3, 3);
 
	ctx.fillStyle = "#1f2a44";
	ctx.fillRect(px + 9, py + 27, 5, 3);
	ctx.fillRect(px + 18, py + 27, 5, 3);
  }
 
  function drawWorld() {
	ctx.clearRect(0, 0, canvas.width, canvas.height);
 
	for (let y = 0; y < rows; y++) {
  	for (let x = 0; x < cols; x++) {
    	drawBlock(x, y, world[y][x]);
  	}
	}
 
	drawPlayer();
  }
 
  function updateInventory() {
    inventoryDiv.innerHTML = "";
    selectedBlockLabel.textContent = blocks[selectedBlock].name;
 
    Object.keys(inventory).forEach(function (type) {
  	const slot = document.createElement("div");
  	slot.className = "slot" + (type === selectedBlock ? " selected" : "");
      slot.dataset.type = type;
 
  	slot.innerHTML = `
    	<div class="swatch" style="background:${blocks[type].color}"></div>
        ${blocks[type].name}: ${inventory[type]}
  	`;
 
      slot.addEventListener("click", function () {
    	selectedBlock = type;
        updateInventory();
  	});
 
      inventoryDiv.appendChild(slot);
	});
  }
 
  function movePlayer(dx, dy) {
	if (!gameStarted) return;
 
	const nextX = player.x + dx;
	const nextY = player.y + dy;
 
	if (
  	nextX < 0 ||
  	nextX >= cols ||
  	nextY < 0 ||
  	nextY >= rows
	) {
  	return;
	}
 
	const target = world[nextY][nextX];
 
	if (blocks[target].solid) {
  	if (inventory[target] !== undefined) {
        inventory[target]++;
  	}
 
      world[nextY][nextX] = "air";
      updateInventory();
  	drawWorld();
  	return;
	}
 
	player.x = nextX;
	player.y = nextY;
	drawWorld();
  }
 
  function placeBlock(tileX, tileY) {
	if (!gameStarted) return;
 
	if (
  	tileX < 0 ||
  	tileX >= cols ||
  	tileY < 0 ||
  	tileY >= rows
	) {
  	return;
	}
 
	if (tileX === player.x && tileY === player.y) {
  	return;
	}
 
	const distance = Math.abs(tileX - player.x) + Math.abs(tileY - player.y);
 
	if (distance > 1) {
  	return;
	}
 
	if (world[tileY][tileX] !== "air") {
  	return;
	}
 
	if (!inventory[selectedBlock] || inventory[selectedBlock] <= 0) {
  	return;
	}
 
    world[tileY][tileX] = selectedBlock;
    inventory[selectedBlock]--;
 
	updateInventory();
	drawWorld();
  }
 
  document.addEventListener("keydown", function (event) {
	const key = event.key.toLowerCase();
 
	if (["w", "a", "s", "d"].includes(key)) {
      event.preventDefault();
	}
 
	if (key === "w") movePlayer(0, -1);
	if (key === "s") movePlayer(0, 1);
	if (key === "a") movePlayer(-1, 0);
	if (key === "d") movePlayer(1, 0);
  });
 
  canvas.addEventListener("contextmenu", function (event) {
    event.preventDefault();
 
	const rect = canvas.getBoundingClientRect();
	const scaleX = canvas.width / rect.width;
	const scaleY = canvas.height / rect.height;
 
	const mouseX = (event.clientX - rect.left) * scaleX;
	const mouseY = (event.clientY - rect.top) * scaleY;
 
	const tileX = Math.floor(mouseX / tileSize);
	const tileY = Math.floor(mouseY / tileSize);
 
	placeBlock(tileX, tileY);
  });
 
  joinBtn.addEventListener("click", function () {
	const chosenName = playerNameInput.value.trim();
 
	player.name = chosenName || "Player";
	gameStarted = true;
 
    startMenu.style.display = "none";
    gameArea.style.display = "block";
 
	drawWorld();
  });
 
  function exitToBlackScreen() {
	gameStarted = false;
    blackScreen.style.display = "block";
  }
 
  exitBtn.addEventListener("click", exitToBlackScreen);
  gameExitBtn.addEventListener("click", exitToBlackScreen);
 
  createWorld();
  updateInventory();
  drawWorld();
})();
